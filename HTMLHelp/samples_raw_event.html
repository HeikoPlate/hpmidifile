Raw Event: Samples 


int time, result;
unsigned char *event_data;
int length;

// Read the event data of a MIDI event

result = HP_ReadEventData(mf,247,&time,&event_data,&length);
if (result != HP_ERR_NONE)
{
   MessageBox(HP_ErrText(result),NULL,MB_OK);
   return;
}

CString r;
int i;

r.Format("Raw Event: time %x ",time);
MessageBox(r,NULL,MB_OK);

for (i=0; i<length; i++)
{
    r.Format("event_data[%i]: %x",i,event_data[i]);
    MessageBox(r,NULL,MB_OK);
}
HP_Delete(event_data);

/* ============================================================================== */

// Insert MIDI event by its data

unsigned char at_data[] = {0xa4,0x63,0x65}; // for example: An Aftertouch event
result = HP_InsertRawEvent(mf,0x3c5,at_data,3);
if (result != HP_ERR_NONE)
{
   MessageBox(HP_ErrText(result),NULL,MB_OK);
   return;
}
