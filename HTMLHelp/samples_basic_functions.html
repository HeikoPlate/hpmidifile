Basic Functions: Samples 


MessageBox(HP_GetVersion(),NULL,MB_OK);


HP_MIDIFILE_MODULE* mf;
UINT result;

mf = HP_Init();
if (mf == NULL)
{
   MessageBox("Err HP_Init",NULL,MB_OK);
   return;
}

result = HP_Load(mf,"c:\\tmp\\test.mid");
if (result!=HP_ERR_NONE)
{
   MessageBox(HP_ErrText(result),NULL,MB_OK);
   return;
}

result = HP_MoreChunks(mf);
if (result == 0)
{
   MessageBox("Only SMF chunks),NULL,MB_OK);
}
else
{
   MessageBox("Any more chunks behind SMF",NULL, MB_OK);
}

result = HP_Save(mf,"c:\\tmp\\new.mid",HP_SMF1);
if (result != HP_ERR_NONE)
{
   MessageBox(HP_ErrText(result),NULL,MB_OK);
   return;
}

result = HP_Free(mf);
if (result != HP_ERR_NONE)
{
   MessageBox(HP_ErrText(result),NULL,MB_OK);
   return;
}
